   public static InputStream getInputStreamFromServer(String uri) {

       HttpEntity entity;
       try {

           entity = HttpUtils.getEntity(uri, null, HttpUtils.METHOD_GET);
           InputStream inputStream = HttpUtils.getInputStream(entity);
           return inputStream;
       } catch (IOException e) {
           e.printStackTrace();
       }
       return null;
   }
